########################################################################
# Setup library
########################################################################
include(GrPlatform) #define LIB_SUFFIX

include_directories(${Boost_INCLUDE_DIR})
link_directories(${Boost_LIBRARY_DIRS})

list(APPEND AD2_AnalogIn_Record_sources
)

set(AD2_AnalogIn_Record_sources "${AD2_AnalogIn_Record_sources}" PARENT_SCOPE)
if(NOT AD2_AnalogIn_Record_sources)
	MESSAGE(STATUS "No C++ sources... skipping lib/")
	return()
endif(NOT AD2_AnalogIn_Record_sources)

add_library(gnuradio-AD2_AnalogIn_Record SHARED ${AD2_AnalogIn_Record_sources})
target_link_libraries(gnuradio-AD2_AnalogIn_Record ${Boost_LIBRARIES} ${GNURADIO_ALL_LIBRARIES})
set_target_properties(gnuradio-AD2_AnalogIn_Record PROPERTIES DEFINE_SYMBOL "gnuradio_AD2_AnalogIn_Record_EXPORTS")

if(APPLE)
    set_target_properties(gnuradio-AD2_AnalogIn_Record PROPERTIES
        INSTALL_NAME_DIR "${CMAKE_INSTALL_PREFIX}/lib"
    )
endif(APPLE)

########################################################################
# Install built library files
########################################################################
include(GrMiscUtils)
GR_LIBRARY_FOO(gnuradio-AD2_AnalogIn_Record RUNTIME_COMPONENT "AD2_AnalogIn_Record_runtime" DEVEL_COMPONENT "AD2_AnalogIn_Record_devel")

########################################################################
# Build and register unit test
########################################################################
include(GrTest)

include_directories(${CPPUNIT_INCLUDE_DIRS})

list(APPEND test_AD2_AnalogIn_Record_sources
    ${CMAKE_CURRENT_SOURCE_DIR}/test_AD2_AnalogIn_Record.cc
    ${CMAKE_CURRENT_SOURCE_DIR}/qa_AD2_AnalogIn_Record.cc
)

add_executable(test-AD2_AnalogIn_Record ${test_AD2_AnalogIn_Record_sources})

target_link_libraries(
  test-AD2_AnalogIn_Record
  ${GNURADIO_RUNTIME_LIBRARIES}
  ${Boost_LIBRARIES}
  ${CPPUNIT_LIBRARIES}
  gnuradio-AD2_AnalogIn_Record
)

GR_ADD_TEST(test_AD2_AnalogIn_Record test-AD2_AnalogIn_Record)

########################################################################
# Print summary
########################################################################
message(STATUS "Using install prefix: ${CMAKE_INSTALL_PREFIX}")
message(STATUS "Building for version: ${VERSION} / ${LIBVER}")
